/*
1.安装git
yum install git 

2.配置git账户
git config --global user.name "your name"
git config --global user.email "your email"

3.创建本地目录
mkdir learngit
cd learngit
4.初始化本地git
git init

5.上传本地文件到配置库
touch readme.txt
git add readme.txt
git commit -m "write a learngit file"

6.查看git状态
git status

7.查看修改的内容
git diff

8.撤出暂存区
git reset HEAD readme.txt

9.删除工作区的改动
git checkout -- readme.txt

10.查看日志
git log
git log --pretty=oneline

11.版本回退
git reset --hard HEAD^
或
git reset --hard commit id

12.删除
git rm readme.txt
git commit -m "delete readme.txt"
 */1:
